/* script */
//탭 메뉴
var $tap_list = $(".tab_menu");

$tap_list.find("ul ul").hide();
$tap_list.find("li.active > ul").show();

function tabMenu(e){
    e.preventDefault();
    var $this = $(this);
    $this.next("ul").show().parent("li").addClass("active")
    .siblings("li").removeClass("active").find(">ul").hide();
}
$tap_list.find("ul>li>a").click(tabMenu).focus(tabMenu);
